  #elem {
    color: #006;
    background-color: #aaa;
    font-size: 25px;

  }
  </style>

<button id="toggle">animate colors</button>
 
<script>
$( "#toggle" ).click(function() {
  $( "#elem" ).animate({
    color: "green",
    backgroundColor: "rgb( 20, 20, 20 )"
  });
});
</script>
 
</body>
